    <script>

        // 1. 頁面切換控制
        function showPage(pageId, btn) {
            // 隱藏所有頁面
            document.querySelectorAll('.page-content').forEach(p => p.classList.remove('page-active'));
            // 顯示目標頁面
            document.getElementById(pageId).classList.add('page-active');
            
            // 更新按鈕樣式
            document.querySelectorAll('.nav-btn').forEach(b => {
                b.classList.remove('nav-link-active', 'text-emerald-500');
                b.classList.add('text-slate-500');
            });
            btn.classList.add('nav-link-active');
            btn.classList.remove('text-slate-500');
            window.scrollTo(0, 0);
        }

        // 2. 複利計算功能
        function calculate() {
            const m = parseFloat(document.getElementById('monthly').value);
            const r = parseFloat(document.getElementById('rate').value) / 100 / 12;
            const y = parseFloat(document.getElementById('years').value) * 12;
            
            if (isNaN(m) || isNaN(r) || isNaN(y)) return;
            
            const res = m * ((Math.pow(1 + r, y) - 1) / r) * (1 + r);
            
            const resultBox = document.getElementById('calc-result');
            resultBox.classList.remove('hidden');
            document.getElementById('result-value').innerText = `$${Math.round(res).toLocaleString()}`;
        }

        // 3. 聊天機器人模擬邏輯
        function toggleChat() {
            document.getElementById('chat-box').classList.toggle('hidden');
        }

        function sendMsg() {
            const input = document.getElementById('chat-input');
            const container = document.getElementById('chat-msgs');
            if (!input.value.trim()) return;

            // 用戶訊息
            const userDiv = document.createElement('div');
            userDiv.className = "bg-emerald-500 text-white p-3 rounded-2xl rounded-tr-none shadow-sm ml-8 text-right";
            userDiv.innerText = input.value;
            container.appendChild(userDiv);
            
            const userMsg = input.value;
            input.value = '';
            container.scrollTop = container.scrollHeight;

            // 模擬機器人回覆
            setTimeout(() => {
                const botDiv = document.createElement('div');
                botDiv.className = "bg-white p-3 rounded-2xl rounded-tl-none shadow-sm border border-slate-100 mr-8";
                
                if (userMsg.includes('0050')) {
                    botDiv.innerText = "0050 是台灣最老牌的 ETF，雖然管理費較 006208 高，但規模大、流動性好，非常適合大資金配置喔！";
                } else if (userMsg.includes('算')) {
                    botDiv.innerText = "你可以點擊上方「複利計算」選單，輸入你的計畫，我會幫你算出未來的資產喔！";
                } else {
                    botDiv.innerText = "這是一個好問題！對於新手，我建議從「定期定額」市值型 ETF 開始，你可以先去「標的比對」頁面看看差異。";
                }
                
                container.appendChild(botDiv);
                container.scrollTop = container.scrollHeight;
            }, 800);
        }

        // 綁定 Enter 鍵送出訊息
        document.getElementById('chat-input').addEventListener('keypress', function (e) {
            if (e.key === 'Enter') sendMsg();
        });
    </script>
